# Chargement du dataset
df = pd.read_csv("datasets/amazon_review.csv")

# Nettoyage, minuscule et suppression ponctuation, chiffres, stopwords
stop_words = set(nltk.corpus.stopwords.words('english'))

def preprocess_text(text):
    text = text.lower()
    text = re.sub(r'[^\w\s]', '', text)  # supprime ponctuation
    text = re.sub(r'\d+', '', text)      # supprime chiffres
    return text

df['clean_text'] = df['reviewText'].apply(preprocess_text)

# Fonction pour convertir tag Treebank -> WordNet (pour lemmatization)
def get_wordnet_pos(treebank_tag):
    if treebank_tag.startswith('J'):
        return 'a'  # adjectif
    elif treebank_tag.startswith('V'):
        return 'v'  # verbe
    elif treebank_tag.startswith('N'):
        return 'n'  # nom
    elif treebank_tag.startswith('R'):
        return 'r'  # adverbe
    else:
        return 'n'  # défaut: nom

# Lemmatisation avec POS tagging + filtrage stopwords
def lemmatize_text(text):
    blob = TextBlob(text)
    lemmas = []
    for word, tag in blob.tags:
        if word in stop_words or len(word) < 2:
            continue
        pos = get_wordnet_pos(tag)
        lemma = Word(word).lemmatize(pos)
        lemmas.append(lemma)
    return lemmas

df['lemmas'] = df['clean_text'].apply(lemmatize_text)

# --- Sentiment Analysis avec TextBlob ---
def get_sentiment(text):
    blob = TextBlob(text)
    polarity = blob.sentiment.polarity
    if polarity > 0.1:
        return 'positive'
    elif polarity < -0.1:
        return 'negative'
    else:
        return 'neutral'

df['textblob_sentiment'] = df['reviewText'].apply(get_sentiment)

# Visualisation de la distribution des sentiments
sentiment_counts = df['textblob_sentiment'].value_counts()

plt.figure(figsize=(8,6))
sns.barplot(x=sentiment_counts.index, y=sentiment_counts.values, palette=['green','red','blue'])
plt.title('Distribution des sentiments détectés par TextBlob')
plt.xlabel('Sentiment')
plt.ylabel('Nombre d\'avis')
plt.show()

pos_counter = Counter()

for text in df['clean_text']:
    blob = TextBlob(text)
    for _, tag in blob.tags:
        if tag.startswith('N'):
            pos_counter['Nouns'] += 1
        elif tag.startswith('V'):
            pos_counter['Verbs'] += 1
        elif tag.startswith('J'):
            pos_counter['Adjectives'] += 1
